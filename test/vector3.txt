class Vector3
{
	var x;
	var y;
	var z;

	func Vector3(_x, _y, _z)
	{
		x = _x;
		y = _y;
		z = _z;
	}

	operator +(v)
	{
		return Vector3(x + v.x, y + v.y, z + v.z);
	}

	operator -(v)
	{
		return Vector3(x - v.x, y - v.y, z - v.z);
	}

	operator *(s)
	{
		return Vector3(x * s, y * s, z * s);
	}

	operator /(s)
	{
		var inv = 1.0 / s;
		return Vector3(x * inv, y * inv, z * inv);
	}

	operator ==(v)
	{
		return x == v.x && y == v.y && z == v.z;
	}

	func string()
	{
		return "(" + x + ", " + y + ", " + z + ")";
	}
}

var a = Vector3(1, 2, 3);
var b = Vector3(-3, -2, -1);
println(a);
println(b);

println("Object: " + a);

println(a + b);
println(a - b);
println(a * 2);
println(a / 2);
