class Vector3
{
	var x;
	var y;
	var z;

	func Vector3(_x, _y, _z)
	{
		x = _x;
		y = _y;
		z = _z;
	}

	operator +(v)
	{
		x += v.x;
		y += v.y;
		z += v.z;
	}

	func string()
	{
		return "(" + x + ", " + y + ", " + z + ")";
	}
}

var a = Vector3(1, 2, 3);
var b = Vector3(-3, -2, -1);
println(a);
println(b);

println("Object: " + a);

a + b;
println(a);
